function getLetter(s) {
  let letter;
  let w = s.charAt(0);
  switch (w) {
    case "a":
      letter = "A";
      break;
    case "e":
      letter = "A";
      break;
    case "i":
      letter = "A";
      break;
    case "o":
      letter = "A";
      break;
    case "u":
      letter = "A";
      break;
    case "b":
      letter = "B";
      break;
    case "c":
      letter = "B";
      break;
    case "d":
      letter = "B";
      break;
    case "f":
      letter = "B";
      break;
    case "g":
      letter = "B";
      break;
    case "h":
      letter = "C";
      break;
    case "j":
      letter = "C";
      break;
    case "k":
      letter = "C";
      break;
    case "l":
      letter = "C";
      break;
    case "m":
      letter = "C";
      break;
    default:
      letter = "D";
  }
  // Write your code here

  return letter;
}
