# Detect vowel and consonent

```jsx
const solution = (letter) => {
  //Write Your solution Here
  // a, e, i, o, u dont forget to return
  if (
    letter === "a" ||
    letter === "e" ||
    letter === "i" ||
    letter === "o" ||
    letter === "u"
  ) {
    return "VOWEL";
  } else {
    return "CONSONANT";
  }
};
```
